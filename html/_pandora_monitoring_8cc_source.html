500TEveElement *PandoraMonitoring::VisualizeCaloHits(const CaloHitList *const pCaloHitList, const std::string &name, TEveElement *parent, const Color color)
501{
502 this->InitializeEve();
503
504 const std::string starter("---");
505 const std::string caloHitListTitle(name.empty() ? "CaloHits" : name);
506 std::string caloHitListName(caloHitListTitle);
507 if (caloHitListName.find(starter) != std::string::npos)
508 caloHitListName.replace(caloHitListName.find(starter), starter.length(), "CaloHits/");
509 std::replace(caloHitListName.begin(), caloHitListName.end(), '\n', '/');
510
511 bool isU(false), isV(false), isW(false);
512
513 if (!pCaloHitList->empty())
514 {
515 HitType hitType(pCaloHitList->front()->GetHitType());
516
517 if (TPC_VIEW_U == hitType)
518 {
519 isU = true;
520 }
521 else if (TPC_VIEW_V == hitType)
522 {
523 isV = true;
524 }
525 else if (TPC_VIEW_W == hitType)
526 {
527 isW = true;
528 }
529 }
530
531 TEveElement *pCaloHitVectorElement = new TEveElementList();
532 pCaloHitVectorElement->SetElementNameTitle(caloHitListName.c_str(), caloHitListTitle.c_str());
533
534 TEveBoxSet *hits = new TEveBoxSet(name.c_str());
535 hits->Reset(TEveBoxSet::kBT_FreeBox, kTRUE, 64);
536 hits->SetOwnIds(kTRUE);
537 hits->SetPickable(kTRUE);
538 hits->SetMainTransparency(5);
539 hits->SetAntiFlick(kTRUE);
540 pCaloHitVectorElement->AddElement(hits);
541
542 static Double_t r[] = {1., 1.}, g[] = {1., 0.}, b[] = {0., 0.}, stop[] = {0., 1.};
543 static const Int_t firstIndex(TColor::CreateGradientColorTable(2, stop, r, g, b, 256));
544
545 Int_t customPalette[256];
546 for (int index = 0; index < 256; ++index)
547 customPalette[index] = firstIndex + index;
548
549 PandoraMonitoringApi::PdgCodeToEnergyMap pdgCodeToEnergyMap;
550 unsigned int firstLayer = std::numeric_limits<unsigned int>::max();
551 unsigned int lastLayer = 0;
552
553 float energySumElectromagnetic = 0.f;
554 float energySumHadronic = 0.f;
555
556 static int colorIter = RED;
557 if (++colorIter >= AUTO)
558 colorIter = RED;
559
560 try
561 {
562 for (const CaloHit *const pCaloHit : *pCaloHitList)
563 {
564 HitType hitType(pCaloHit->GetHitType());
565
566 if ((isU && TPC_VIEW_U != hitType) || (isV && TPC_VIEW_V != hitType) || (isW && TPC_VIEW_W != hitType))
567 throw StatusCodeException(STATUS_CODE_FAILURE);
568
569 // Determing extremal pseudolayers
570 const unsigned int pseudoLayer(pCaloHit->GetPseudoLayer());
571
572 if (pseudoLayer > lastLayer)
573 lastLayer = pseudoLayer;
574
575 if (pseudoLayer < firstLayer)
576 firstLayer = pseudoLayer;
577
578 // Energy properties
579 const float hitEnergy(pCaloHit->GetElectromagneticEnergy());
580 energySumElectromagnetic += hitEnergy;
581
582 const float hitEnergyHadronic(pCaloHit->GetHadronicEnergy());
583 energySumHadronic += hitEnergyHadronic;
584
585 // MC particle id
586 int particleId = 0;
587
588 try
589 {
590 const MCParticle *pMCParticle(MCParticleHelper::GetMainMCParticle(pCaloHit));
591 particleId = pMCParticle->GetParticleId();
592 }
593 catch (StatusCodeException &)
594 {
595 }
596
597 PandoraMonitoringApi::PdgCodeToEnergyMap::iterator it = pdgCodeToEnergyMap.find(particleId);
598
599 if (pdgCodeToEnergyMap.end() == it)
600 {
601 pdgCodeToEnergyMap.insert(std::make_pair(particleId, hitEnergy));
602 }
603 else
604 {
605 float oldValue = it->second;
606 it->second = oldValue + hitEnergy;
607 }
608
609 // Supply hit marker details
610 EColor hitColor = GetROOTColor(color);
611
612 if (color == AUTOID)
613 {
614 hitColor = GetROOTColor(GetColorForPdgCode(particleId));
615 }
616 else if (color == AUTOITER)
617 {
618 hitColor = GetROOTColor(Color(colorIter));
619 }
620 else if (color == AUTOENERGY)
621 {
622 unsigned int customColorIndex = 0;
623
624 if (m_energyScaleThresholdE > 0.f)
625 customColorIndex = std::min(255, static_cast<int>(255.f * (hitEnergy / m_energyScaleThresholdE)));
626
627 hitColor = EColor(customPalette[customColorIndex]);

1005//------------------------------------------------------------------------------------------------------------------------------------------
1006
1007bool MakePathIfNeeded(const std::string& path)
1008{
1009 // ATTN: mkdir used here, not stat to avoid race conditions.
1010 mode_t mode = 0755;
1011 int ret = mkdir(path.c_str(), mode);
1012
1013 if (ret == 0)
1014 return true;
1015
1016 switch (errno)
1017 {
1018 // ATTN: The parent doesn't exist, so try and make that folder first by recursing.
1019 case ENOENT:
1020 {
1021 int pos = path.find_last_of('/');
1022 if (pos == std::string::npos)
1023 return false;
1024
1025 if (!MakePathIfNeeded(path.substr(0, pos)))
1026 return false;
1027 }
1028 return (0 == mkdir(path.c_str(), mode));
1029
1030 // ATTN: There is already something there, so check it is a folder.
1031 case EEXIST:
1032 return CheckPathExists(path);
1033
1034 // ATTN: In any other case, assume folder creation failed.
1035 // This includes too long a path, disk quotas and more.
1036 default:
1037 return false;
1038 }
1039}

1224{
1225 if (symmetryOrder > 2)
1226 {
1227 static const double pi(std::acos(-1.));
1228 const double x0(-1. * closestDistanceToIp * tan(pi / double(symmetryOrder)));
1229 const double y0(closestDistanceToIp);
1230
1231 for (int i = 0; i < symmetryOrder; ++i)
1232 {
1233 double theta = 0.f;
1234 theta = phi0 + (2 * pi * double(i) / double(symmetryOrder));
1235
1236 double x = x0 * cos(theta) + y0 * sin(theta);
1237 double y = y0 * cos(theta) - x0 * sin(theta);
1238 coordinates.push_back(std::pair<double, double>(x, y));
1239 }
1240 }

1278 const int nvertices(vertices.size());

1280 double *y = new double[nvertices];
1281
1282 int index = 0;
1283 for (DoublePairVector::iterator itCoord = vertices.begin(), itCoordEnd = vertices.end(); itCoord != itCoordEnd; ++itCoord)
1284 {
1285 x[index] = (*itCoord).first;
1286 y[index] = (*itCoord).second;
1287 ++index;
1288 }
1289
1290 TGeoXtru *pTGeoXtru = new TGeoXtru(2);
1291
1292 pTGeoXtru->DefinePolygon(nvertices,x,y);
1293 double z0 = -halfLength, x0 = 0.0, y0 = 0.0;
1294 double z1 = halfLength, x1 = 0.0, y1 = 0.0;
1295
1296 double scale0 = 1.0;
1297 pTGeoXtru->DefineSection(0, z0, x0, y0, scale0); // Z position, offset and scale for first section
1298 pTGeoXtru->DefineSection(1, z1, x1, y1, scale0); // -''- go forward
1299
1300 delete[] x;
1301 delete[] y;
1302
1303 return pTGeoXtru;
1304}
